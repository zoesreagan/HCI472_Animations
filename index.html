/* jQuery animation class */

.circle-1-animate {
    animation-name: squash;

/* CSS keyframe animation */

@-webkit-keyframes bounce {
  0% { -webkit-transform: translateY(-300px) translateX(25px); }

  40% { -webkit-transform: translateY(0px); }

  50% { -webkit-transform: translateY(-200px);  }

  60% { -webkit-transform: translateY(0px); }

  75% { -webkit-transform: translateY(-100px);  }

  90% { -webkit-transform: translateY(0px); }

  96% { -webkit-transform: translateY(-50px); }

  100% { -webkit-transform: translateY(0px);  }
  }

/* jQuery animation class */

.circle-3-animate { animation: bounce 2.5s cubic-bezier(0.175, 0.885, 0.32, 1.275); }
